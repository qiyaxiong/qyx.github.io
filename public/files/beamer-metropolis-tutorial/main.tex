\section{Why Metropolis}

\begin{frame}{Theme selection criteria}
  \begin{columns}[T,onlytextwidth]
    \column{0.48\textwidth}
      \textbf{Good for reuse}
      \begin{itemize}
        \item Low visual noise
        \item Stable default layout
        \item Works well with equations and code
      \end{itemize}

    \column{0.48\textwidth}
      \textbf{Easy to share}
      \begin{itemize}
        \item Available in common TeX distributions
        \item Easy to try in Overleaf
        \item Needs little configuration
      \end{itemize}
  \end{columns}
\end{frame}

\section{Slide structure}

\begin{frame}{One slide, one claim}
  \begin{block}{Practical rule}
    Put the conclusion in the slide title, then add only the evidence needed to support it.
  \end{block}

  \begin{exampleblock}{Recommended structure}
    Title = claim; body = up to three supporting points; progress bar = pacing.
  \end{exampleblock}
\end{frame}
